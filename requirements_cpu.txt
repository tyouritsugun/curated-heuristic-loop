# CHL API Server - CPU-only mode (no ML dependencies)
#
# This requirements file is for the API server running in CPU-only mode.
# In this mode, search uses SQLite text search (LIKE queries) instead of
# vector embeddings and semantic similarity.
#
# Installation:
#   python -m venv .venv-cpu
#   source .venv-cpu/bin/activate  # On Windows: .venv-cpu\Scripts\activate
#   python -m pip install --upgrade pip
#   python -m pip install -r requirements_cpu.txt
#
# Starting the API server:
#   python -m uvicorn src.api.server:app --host 127.0.0.1 --port 8000
#   (Backend is automatically configured from data/runtime_config.json)
#
# Note: The MCP server is installed separately via `uv sync` from pyproject.toml
# and communicates with this API server via HTTP.
